3 Phases"
1. Initialization. Evaluates settings.gradle.kts. Creates Settings and Project for every project
2. Configuration. Evaluates all builds scripts build.gradle.kts. Creates tasks graph.
3. Execution. Schedules and executes the selected tasks.

settings.gradle.kts:
 rootProject.name = "basic"
 println("This is executed during the initialization phase.")

build.gradle.kts:
  println("This is executed during the configuration phase.")

  tasks.register("test") {
      doLast {
          println("This is executed during the execution phase.")
      }
      println("This is executed during the configuration phase, only when the "test" task is requested.")
  }
